Como exibir apenas alunos aprovados?

Filtro


<li ng-repeat="alu in alunos.nomes">{{ alu.nome | uppercase }}</li>
<li ng-repeat="alu in alunos.nomes">{{ alu.nome | lowercase }}</li>
<li ng-repeat="alu in alunos.nomes | filter:{nome: 'art'}">
{{ alu.nome }}</li>

<li ng-repeat="alu in alunos.nomes | limitTo:2">{{ alu.nome }}</li>

Ordenação


<li ng-repeat="alu in alunos.nomes | orderBy:'nome'">
{{ alu.nome | uppercase }}</li>

Ascendente


<li ng-repeat="alu in alunos.nomes | orderBy:'-nome'">

Factory e Service

  • Encapsulamento de funcionalidades.
  • Geralmente utilizado para funcionalidades que se repetem.
  • Factory: Escrevemos o objeto a ser retornado dentro dele
  • Service: Usamos o this

Service

meuApp.service('meuService', function(){
  //var this = {}
  this.listaDeTarefas = [];
  this.adicionaTarefa = function(tarefa){
    //Faz algo
  }
  this.removeTarefa = function(tarefa){
    //Faz algo
  }
  //return this;
});

Factory

meuApp.factory('meuFactory', function(){
  var factory = {}
  listaDeTarefas = [];
  factory.adicionaTarefa = function(tarefa){
    //Faz algo
  }
  factory.removeTarefa = function(tarefa){
    //Faz algo
  }
  return factory;
});

Factory

meuApp.factory('Tarefa', function(){
  return function(descricao, prioridade){
    return{
      descricao: descricao,
      prioridade: prioridade,
      completo: function(){
        //Fazer algo quando completo
      }
    }
  }
});

Component

Novo no AngularJS 1.5

// Antes
app.directive(name, fn)
 
// Agora
app.component(name, options)
app.directive('listDirective', function () {
 return {
     scope: { // Scope isolado
         message: '@'
     },
     template: '
Hello {{$ctrl.message}}
',
     controller: function () {
             this.message = "Arthur directive"
     },
     //defaults, automaticamente setado quando usa .component()
     controllerAs: '$ctrl',
     bindToController: true
 };
});
app.component('listComponent', {
 bindings: { // scope isolado
     message: '@'
 },
 template:'
Hello {{$ctrl.message}}
',
 controller: function () {
     this.message = "Arthur component"
 }
});

No HTML

     message: '@'
},

         message: '@'
     },

No lugar de @, pode-se usar:

  • = Two-way data binding: atualiza nos dois sentidos
  • < One-way data binding: quando apenas quer ler
  • @ Para strings
  • & Executa expressões
